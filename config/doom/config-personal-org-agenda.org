(defun cashpw/org-agenda-files (context &optional include-archive)
  "Return list of agenda files for CONTEXT. Include archived files when INCLUDE-ARCHIVE."
  (let ((files
         (cond
          ((equal context 'notes-with-todo)
           (cashpw/notes-files-with-tag "hastodo"))
          ((equal context 'notes-all)
           (cashpw/org-agenda-files--notes-all))
          ((equal context 'personal)
           (append
            `(,(cashpw/path-todos)
              ,cashpw/path--personal-asana
              ,(cashpw/path-calendar))))
          ((equal context 'projects-with-todo)
           (cashpw/notes-files-with-tags "hastodo" "project"))
          ((equal context 'pets-with-todo)
           (cashpw/notes-files-with-tags "hastodo" "pet"))
          ((equal context 'calendar)

          ((equal context 'journal-this-year)
           (cashpw/org-agenda-files--journal-this-year))
          ((equal context 'journal-this-year-with-todo)
           (cashpw/org-agenda-files--journal-this-year-with-todo))

           (cashpw/notes-files-with-tags "person" "private")))))
    (if include-archive
        (append
         files
         (org-extras-files-archive-files files))
      files)))

** Method

I continue to iterate on this TODO management method which is a combination of [[wikipedia:Getting_Things_Done][Getting Things Done]] and what I've found that works for me.

TODOs come from (1) captures and (2) my calendar via =org-gcal=. TODOs are in the Inbox unless they meet one of the following criteria:

1. Tagged with =someday=

   Example: Read a book a friend recommended to me.

2. Tagged with =reminder=

   Example: Someone's birthday.

3. They are 'specified'; they have the following properties:

   - An =Effort= of less than four hours
   - An explicit =CATEGORY=
   - A start date/time in =SCHEDULED=
   - (optional) An end date/time in =DEADLINE=

  "Wrapper for `org-get-title' that is safe to call in non-org buffers."
  (let ((buf (cond ((bufferp buffer-or-file) buffer-or-file)
                   ((stringp buffer-or-file) (find-file-noselect buffer-or-file))

          (org-get-title)
        (or (bound-and-true-p category)
            "")))))

(defconst cashpw-org-agenda-week-tag-regexp "^[0-9]\\{4\\}w[0-9]\\{2\\}$"
  "Regular expression matching standard zero-padded week tags YYYYwWW.")

(defun cashpw/org-agenda-week-tag-p (tag)
  "Return t if TAG matches the YYYYwWW week format."
  (string-match-p cashpw-org-agenda-week-tag-regexp tag))

(defun cashpw/org-agenda-item-tags (item)
  "Return tags for ITEM (a string representing an agenda line carrying text properties like 'tags or 'org-marker), checking text properties and falling back to marker."
  (or (get-text-property 0 'tags item)
      (org-super-agenda--get-tags item)
      (let ((marker (or (get-text-property 0 'org-marker item)
                        (get-text-property 0 'org-hd-marker item))))
        (when marker
          (org-super-agenda--when-with-marker-buffer marker
            (org-get-tags))))))

(defun cashpw/org-agenda-item-has-week-tag-p (item)
  "Return t if ITEM has any tag matching YYYYwWW format."
  (let ((tags (cashpw/org-agenda-item-tags item)))
    (--any (cashpw/org-agenda-week-tag-p it) tags)))

(defun cashpw/org-agenda-item-future-week-p (item)
  "Return t if ITEM has a week tag strictly greater than next week's tag.
Comparing tags chronologically using `string-greaterp` works only because
the week tags are strictly zero-padded (e.g. \"2026w09\" < \"2026w10\")."
  (let ((next-week (or cashpw-org-agenda-next-week-tag
                       (cashpw-time-next-week-tag)))
        (tags (cashpw/org-agenda-item-tags item)))
    (--any (and (cashpw/org-agenda-week-tag-p it)
                (string-greaterp it next-week))
           tags)))

(defun cashpw/org-agenda-current-line-tags ()

                        (cashpw/org-agenda-get-property-on-line 'org-hd-marker))))
        (when marker
          (org-super-agenda--when-with-marker-buffer marker
            (org-get-tags))))))

*** Effort scanner

These helpers scan and calculate capacity metrics directly from the agenda buffer's text properties, enabling zero-I/O capacity scanning and calculations without opening underlying Org files.

#+begin_src emacs-lisp :tangle config-personal.el :results none
(defun cashpw/org-agenda-get-property-on-line (property)
  "Get the value of text PROPERTY anywhere on the current line.
PROPERTY is typically a symbol representing a text property (e.g. 'effort-minutes).
Return the property value if found, or nil otherwise."
  (save-excursion
    (let* ((beg (line-beginning-position))
           (end (line-end-position))
           (p beg)
           (val nil))
      (while (and (< p end) (not val))
        (setq val (get-text-property p property))
        (unless val
          ;; Efficiently jump to the next position where the property might change.
          (setq p (next-single-property-change p property nil end))))
      val)))

(defun cashpw/org-agenda-calculate-total-effort (&optional predicate)
  "Calculate the total effort (in minutes) by summing the `effort-minutes` property in the current agenda buffer.
If PREDICATE is non-nil, only sum effort for lines where PREDICATE returns true.
PREDICATE is called with no arguments at the beginning of each line."
  (save-excursion
    (goto-char (point-min))
    (let ((total 0))
      (while (not (eobp))
        (let ((effort (cashpw/org-agenda-get-property-on-line 'effort-minutes)))
          (when (and (numberp effort)
                     (or (null predicate)
                         (save-excursion (funcall predicate))))
            (setq total (+ total effort))))
        (forward-line 1))
      total)))

*** Capacity Header

The Capacity Header feature dynamically calculates and injects daily planned
capacity metrics at the top of the custom Day Planner view buffer.

It works in a zero-I/O fashion by scanning the finalized agenda buffer lines to
extract the cached `effort-minutes` text properties, rather than reading files
or querying org markers. To prevent cumulative double-renders when rebuilding or
refreshing the agenda view, it utilizes a robust teardown mechanism that locates
and removes any existing capacity header blocks by looking for a custom text
property (`'cashpw-header t`).

To prevent unhandled runtime exceptions from crashing the Org Agenda interface,
the hook is registered using a safe wrapper function bounded by `condition-case`.

#+begin_src emacs-lisp :tangle config-personal.el :results none
(defcustom cashpw-org-agenda-daily-capacity-hours 6.0
  "Default daily planned capacity limit in hours (e.g. 6.0 hours is 80% of an 8-hour workday)."
  :type 'float
  :group 'cashpw)

(defcustom cashpw-org-agenda-weekly-capacity-hours 30.0
  "Default weekly planned capacity limit in hours."
  :type 'float
  :group 'cashpw)

(defun cashpw/org-agenda--insert-capacity-header (label total-minutes capacity-hours &optional detail-suffix)
  "Format and insert planned capacity metrics at the top of the buffer."
  (let* ((total-hours (/ total-minutes 60.0))
         (percentage (if (> capacity-hours 0.0)
                         (* 100 (/ total-hours capacity-hours))
                       0.0))
         (header-text (format "Planned %s: %.1fh / %.1fh Capacity (%.0f%%%s)\n\n"
                              label total-hours capacity-hours percentage (or detail-suffix ""))))
    (goto-char (point-min))
    (insert (propertize header-text
                        'face 'org-agenda-structure
                        'read-only t
                        'rear-nonsticky t
                        'cashpw-header t))))

(defun cashpw/org-agenda-display-planned-capacity-header ()
  "Insert planned capacity metrics at the top of planner buffers.
Clears any previous headers to prevent text duplication on buffer refreshes."
  (let ((inhibit-read-only t))
    (save-excursion
      (goto-char (point-min))
      ;; 1. Teardown: Clear only our dynamically injected capacity headers
      (while (re-search-forward "^Planned \\(Today\\|This Week\\):.*\n\n" nil t)
        (when (get-text-property (match-beginning 0) 'cashpw-header)
          (replace-match "")))
      
      ;; 2. Calculation & Injection
      (goto-char (point-min))
      (cond
       ;; Case A: Day Planner
       ((equal org-keys ".today")
        (let ((today-only-pred (lambda () (cashpw/org-agenda-get-property-on-line 'date))))
          (cashpw/org-agenda--insert-capacity-header
           "Today"
           (cashpw/org-agenda-calculate-total-effort today-only-pred)
           cashpw-org-agenda-daily-capacity-hours)))
       
       ;; Case B: Week Planner
       ((equal org-keys ".plan-week")
        (let ((week-only-pred (lambda ()
                                (member (cashpw-time-current-week-tag)
                                        (cashpw/org-agenda-current-line-tags)))))
          (cashpw/org-agenda--insert-capacity-header
           "This Week"
           (cashpw/org-agenda-calculate-total-effort week-only-pred)
           cashpw-org-agenda-weekly-capacity-hours
           " of Weekly Limit")))))))

(defun cashpw/org-agenda-display-planned-capacity-header-safe ()
  "Safely execute capacity header injection, preventing hook errors from breaking agenda."
  (condition-case err
      (cashpw/org-agenda-display-planned-capacity-header)
    (error
     (message "Error displaying planned capacity header: %s" (error-message-string err)))))

(after! org-agenda
  (add-hook 'org-agenda-finalize-hook #'cashpw/org-agenda-display-planned-capacity-header-safe))

*** Backlog triage
Helper functions and keybindings to quickly assign tasks in the agenda to specific weeks.

- `w .` :: Assign entry at point to the current week (`cashpw/org-agenda-assign-to-this-week`)
- `w n` :: Assign entry at point to the next week (`cashpw/org-agenda-assign-to-next-week`)
- `w w` :: Assign entry at point to a custom week (`cashpw/org-agenda-assign-to-week`)

#+begin_src emacs-lisp :tangle config-personal.el :results none
(defun cashpw/org-agenda-update-week-tags (current-tags week-tag)
  "Return CURRENT-TAGS with old week tags removed and WEEK-TAG added if non-nil/non-empty.
WEEK-TAG must match YYYYwWW format if provided."
  (when (and week-tag (not (string-empty-p week-tag))
             (not (string-match-p "^[0-9]\\{4\\}w[0-9]\\{2\\}$" week-tag)))
    (user-error "Invalid week tag format: %s. Expected YYYYwWW" week-tag))
  ;; Remove any existing week tags matching YYYYwWW format (e.g., 2026w21)
  (let ((clean-tags (--remove (string-match-p "^[0-9]\\{4\\}w[0-9]\\{2\\}$" it) current-tags)))
    (if (and week-tag (not (string-empty-p week-tag)))
        (append clean-tags (list week-tag))
      clean-tags)))

(defun cashpw/org-agenda-assign-to-week (week-tag)
  "Assign the entry at point to the specified WEEK-TAG, cleaning out old week tags."
  (interactive (list (read-string "Week (YYYYwWW): " (cashpw-time-current-week-tag))))
  (let* ((marker (or (get-text-property (point) 'org-marker)
                     (user-error "No entry at point")))
         (buffer (marker-buffer marker))
         (pos (marker-position marker)))
    (unless (and buffer (buffer-live-p buffer))
      (user-error "Marker points to an invalid or killed buffer"))
    (with-current-buffer buffer
      (save-excursion
        (goto-char pos)
        (org-set-tags (cashpw/org-agenda-update-week-tags (org-get-tags nil 'local) week-tag))
        (org-todo-trigger-tag-changes nil)))
    (org-agenda-redo)))

(defun cashpw/org-agenda-assign-to-this-week ()

(defun cashpw/org-clock-duration-in-minutes (&optional arg)
  "Return total clocked minutes.

Based on `org-clock-display'.

By default, show the total time for the range defined in
`org-clock-display-default-range'.  With `\\[universal-argument]' \
ARG prefix, show
the total time for today instead.

With `\\[universal-argument] \\[universal-argument]' ARG prefix, \
use a custom range, entered at prompt.

With `\\[universal-argument] \ \\[universal-argument] \
\\[universal-argument]' ARG prefix, display the total time in the
echo area.

Use `\\[org-clock-remove-overlays]' to remove the subtree times."
  (interactive "P")
  (org-clock-remove-overlays)
  (let* ((todayp (equal arg '(4)))
         (customp (member arg '((16) today yesterday
                                thisweek lastweek thismonth
                                lastmonth thisyear lastyear
                                untilnow interactive)))
         (prop (cond ((not arg) :org-clock-minutes-default)
                     (todayp :org-clock-minutes-today)
                     (customp :org-clock-minutes-custom)
                     (t :org-clock-minutes))))
    (cond ((not arg) (org-clock-sum-custom
                      nil org-clock-display-default-range prop))
          (todayp (org-clock-sum-today))
          (customp (org-clock-sum-custom nil arg))
          (t (org-clock-sum)))
    (unless (equal arg '(64))
      (save-excursion
        (goto-char (point-min))
        (let ((p nil))
          (while (or (and (equal (setq p (point)) (point-min))
                          (get-text-property p prop))
                     (setq p (next-single-property-change (point) prop)))
            (goto-char p)
            (let ((time (get-text-property p prop)))
              (when time (org-clock-put-overlay time)))))
        ;; Arrange to remove the overlays upon next change.
        (when org-remove-highlights-with-change
          (add-hook 'before-change-functions #'org-clock-remove-overlays
                    nil 'local))))
    org-clock-file-total-minutes))

There are three types of actions which can occur when I mark a =TODO= item as done. They have a precedence and I've listed them in that order.

1. =noop=: The item remains (1) in the same file and (2) as a =TODO=.
2. =keep=: The item remains in the same file and the =TODO= state changes to, e.g., =DONE=.
3. =delete=: Delete the item.
4. =archive= (default action): Archive the item.

The return value of hooks determine the action we take. All =noop= hooks must return nil before we consider cutting the entry. Likewise, all =cut= hooks must return nil before we archive the entry.

  "Archive entry when it is marked as done (as defined by `org-done-keywords')."
  (org-clock-out-if-current)

    (asana-task-complete (string-trim-left asana-id ".+-")))

(defun cashpw/org-gcal--timestamp-from-event (event)
  "Return an Org-mode timestamp from Google Calendar EVENT."
  (let* ((start-time (plist-get (plist-get event :start)
                                :dateTime))
         (end-time (plist-get (plist-get event :end)
                              :dateTime))
         (start-day  (plist-get (plist-get event :start)
                                :date))
         (end-day  (plist-get (plist-get event :end)
                              :date))
         (start (if start-time
                    (org-gcal--convert-time-to-local-timezone
                     start-time
                     org-gcal-local-timezone)
                  start-day))
         (end   (if end-time
                    (org-gcal--convert-time-to-local-timezone
                     end-time
                     org-gcal-local-timezone)
                  end-day))
         (old-time-desc (org-gcal--get-time-and-desc))
         (old-start (plist-get old-time-desc :start))
         (old-end (plist-get old-time-desc :start))
         (recurrence (plist-get event :recurrence)))
    ;; Keep existing timestamps for parent recurring events.
    (when (and recurrence old-start old-end)
      (setq start old-start
            end old-end))
    (cashpw/org-gcal--timestamp start end)))

(defun cashpw/org-gcal--timestamp (start end)
  "Return an Org-mode timestamp from ISO START and END times."
  (if (or (string= start end)
          (org-gcal--alldayp start end))
      (org-gcal--format-iso2org start)
    (if (and (= (plist-get (org-gcal--parse-date start) :year)
                (plist-get (org-gcal--parse-date end)   :year))
             (= (plist-get (org-gcal--parse-date start) :mon)
                (plist-get (org-gcal--parse-date end)   :mon))
             (= (plist-get (org-gcal--parse-date start) :day)
                (plist-get (org-gcal--parse-date end)   :day)))
        (format "<%s-%s>"
                (org-gcal--format-date start "%Y-%m-%d %a %H:%M")
                (org-gcal--format-date end "%H:%M"))
      (format "%s--%s"
              (org-gcal--format-iso2org start)
              (org-gcal--format-iso2org
               (if (< 11 (length end))
                   end
                 (org-gcal--iso-previous-day end)))))))

(defun cashpw/org-gcal--start (event)
  "Return the scheduled start time for EVENT.

Reference: `org-gcal--update-entry'."
  (let* ((start (plist-get event :start))
         (start-datetime (plist-get start :dateTime))
         (start-date (plist-get start :date)))
    (if start-datetime
        (parse-iso8601-time-string start-datetime)
      (date-to-time start-date))))

(defun cashpw/org-gcal--end (event)
  "Return the scheduled end time for EVENT.

Reference: `org-gcal--update-entry'."
  (let* ((end (plist-get event :end))
         (end-datetime (plist-get end :dateTime))
         (end-date (plist-get end :date)))
    (if end-datetime
        (parse-iso8601-time-string end-datetime)
      (date-to-time end-date))))

(defun cashpw/org-get-drawer-contents (drawer-name)
  "Return the contents of DRAWER-NAME for the current Org entry."
  (if (not (org-at-heading-p))

                                            (org-element-property :contents-end element))))))))

     (beginning-of-line)
     (point))

     (goto-char end-point)
     (next-line)
     (beginning-of-line)
     (point))))

(defun cashpw/org-gcal--maybe-create-prep-meeting
    (_calendar-id event _update-mode)
  "Insert a prep TODO if there are more than one attendees to the meeting EVENT.
_CALENDAR-ID and _UPDATE-MODE are ignored."
  (when (and (not (org-gcal-extras--processed-p))
             (sequencep event) (> (length (plist-get event :attendees)) 1)
             (--none-p
              (string-match-p it (plist-get event :summary))
              cashpw/org-gcal--no-prep-reminder-summaries))
    (let* ((event-start-time (cashpw/org-gcal--start event))
           (prepare-time
            (cashpw-time--zero-out-hh-mm-ss
             (cond
              ((day-of-week-monday-p event-start-time)
               ;; Prepare for Monday events on the preceding Friday
               (org-time-subtract event-start-time (days-to-time 3)))
              (t
               ;; Prepare for all other events on the day before
               (org-time-subtract event-start-time (days-to-time 1)))))))
      (unless (and
               (cashpw-time-past-p prepare-time)
               (not (cashpw-time-today-p prepare-time)))
        (cashpw/org-gcal--create-prep-meeting
         (plist-get event :summary) prepare-time)))))

(defun cashpw/org-gcal-fetch ()
  "Clear calendar buffer and fetch events."
  (interactive)

  ;; Ignore these methods to improve performance. This is safe
  ;; because I don't push any events to GCal
  (cl-letf (((symbol-function 'org-generid-id-update-id-locations) #'ignore)
            ((symbol-function 'org-gcal-sync-buffer) #'ignore))
    (flyspell-mode 0)

    (org-gcal-sync-tokens-clear)
    (let* ((calendar-path (cdr (car org-gcal-fetch-file-alist)))
           (org-agenda-files
            ;; Set limited agenda files to improve performance
            `(,calendar-path)))
      (deferred:sync! (org-gcal-fetch)))

    (flyspell-mode 1)))

(defun cashpw/org-gcal-clear-and-fetch ()
  "Clear calendar buffer and fetch events."
  (interactive)
  (org-gcal-activate-profile cashpw/org-gcal--profile-personal)
  (let ((calendar-path (cdr (car org-gcal-fetch-file-alist))))
    (with-current-buffer (find-file-noselect calendar-path)
      (when (cashpw/buffer-contains-regexp-p ":LOGBOOK:")
        (error
         "Wait! Calendar contains un-archived LOGBOOK entries. Archive these, then try again."))
      ;; Keep the buffer tidy
      (cashpw/org-remove-past-scheduled-events)

      ;; Delete extra TODOs. We'll re-create these in the fetch if they're still relevant.

** Create TODO

#+begin_src emacs-lisp :tangle config-personal.el :results none
(cl-defun cashpw/org-insert-todo
    (text &key point-or-marker priority effort category start-time end-time include-hh-mm)
  "Insert TODO with TEXT at POINT-OR-MARKER.

Optionally set the TODO's PRIORITY, EFFORT, CATEGORY, START-TIME, and END-TIME.  If INCLUDE-HH-MM is non-nil, include hours and minutes in the schedule timestamp."
  (save-excursion
    (when point-or-marker
      (goto-char point-or-marker))
    (org-insert-todo-heading-respect-content)
    (insert text)
    (quiet!
      (when priority
        (org-priority priority))
      (when effort
        (org-set-property "Effort" effort))
      (when category
        (org-set-property "CATEGORY" category))
      (when (or start-time end-time)
        (cond
         ((and end-time (null start-time))
          (error "Cannot schedule end-time without start-time."))
         ((and start-time end-time)
          (org-schedule
           nil
           (format "%s-%s"
                   (format-time-string "%F %a %H:%M" start-time)
                   (format-time-string "%H:%M" end-time))))
         ((and start-time include-hh-mm)
          (org-schedule nil (format-time-string "%F %a %H:%M" start-time)))
         ((and start-time)
          (org-schedule nil (format-time-string "%F %a" start-time))))))))
#+end_src

(defun cashpw/org-schedule-for-block (block-time &optional date)
  "Schedule the current Org entry for the time range in BLOCK-TIME on DATE."
  (interactive)
  (let ((start-time (plist-get block-time :start))
        (end-time (plist-get block-time :end))
        (date (or date "today")))
    (org-schedule nil (format "%s %s-%s"
                              date
                              start-time
                              end-time))))

(defun cashpw/org-schedule-today-from-to (start-time end-time &optional date)
  "Schedule the current Org entry from START-TIME to END-TIME on DATE."
  (interactive)

    (org-schedule nil (format "%s %s-%s"
                              date
                              start-time
                              end-time))))
